### 相关API
#### 1. 添加人员（仅限管理员）
用于添加访问账户，可选用户所属角色组，关系到访问权限。新添加的用户的密码都是`123456`，并在登录时以`is_simple`标记用户是否修改了默认密码。
```javascript
{
    url : "127.0.0.1:7999/api/accounts"
    method : POST
    data : {
        "username"  # 用户名
        "role"      # 角色(粮油公司0, 教体局1, 学校2)
    }
    return : {
        "msg":     # 消息
        "data":    # 数据
        "code":    # 状态码
    }
}
```
示例：
```javascript
{
    url : "http://127.0.0.1:7999/api/accounts/"
    method : POST
    data : {
        "username" : "ldzx",
        "role" : 2
    }
    return : {
        "msg": "Created successfully",
        "data": {
            "id": 8,
            "username": "ldzx",
            "is_active": true,
            "role": 2
        },
        "code": 201
    }   
}
```
#### 2. 修改密码（用户）
用于用户修改自己的密码，仅能修改自己的密码
```javascript
{
    url : http://127.0.0.1:8000/api/accounts/<id>/
    method : PUT
    data: {
        "username" # id对应的用户名
        "password" # 修改后的密码
    }
    return : {
        "msg"
        "data":{
            "id"
            "username"
            "is_active"
            "role"
        }
        "code"
    }
}
```
示例
```javascript
{
    url : http://127.0.0.1:8000/api/accounts/4/
    method : PUT
    data : {
        "username":"ldzx",
        "password":"654321"
    }
    return : {
        "msg":"Updated successfully",
        "data":{
            "id": 4,
            "username": "ldzx",
            "is_active": true,
            "role": 2
        },
        "code":200
    }  
}
```

#### 4. 用户登录
用于用户登录获得`token`，系统会返回`token`、`username`、`role`等，可用于前端分权限显示界面。绝大部分后端API都需要传入`token`。
```javascript
{
    url : http://127.0.0.1:7999/api/login/
    method : POST
    data : {
        "username" # 用户名
        "password" # 密码
    }
    return : {
        "msg"
        "data":{
            "refresh"   # 用于刷新token，暂时没用
            "access"    # token，用于权限管理
            "id"        # 用户ID,有很多地方需要传入id
            "username"  # 用户名

            "is_simple" # 标记密码是否为默认密码，如果为    true，应指引用户进行修改
            "is_admin"  # 标记是否为管理员
        },
        "code"
    }
}
```

## 商品相关

#### 1. 查看所有商品
**权限：所有登录的用户**

用于查看所有已上架的商品，当商品目前没有可用的价格时，`price`,`start_time`,`end_time`,`unit`都将被返回为`null`
```javascript
{
    url : http://127.0.0.1:8000/api/goods
    method : GET
    params : {
        page        # 页数
        page_size   # 每页的个数
    }
    return : {
        "msg",
        "data":{
            "count"                  # 查询到的总个数
            "next"                   # 下一页
            "previous"               # 上一页
            "results": [
                {
                    "id"             # 商品ID
                    "name"           # 商品名称
                    "image"          # 商品图片
                    "description"    # 商品描述
                    "unit"           # 商品的计量单位
                    "status"         # 商品状态
                    "price"          # 商品价格
                    "start_time"     # 价格开始时间
                    "end_time"       # 价格结束时间
                }
            ]
        },
        "code"
    }
}
```

**权限：仅粮油公司组(`role=0`)**

```javascript
{

    method : POST

        "name"          # 商品名称
        "image"         # 商品图片
        "description"   # 商品描述
        "price"         # 商品价格
        "unit"          # 商品计量单位（注意需要先在数据库中添加计量单位，然后传入对应单位的id）
        "category"      # 商品类别（注意需要先在数据库中添加商品类别，然后传入对应单位的id）
    }
    return : {
        "msg",
        "data":{
            "id"
            "name"
            "image"         # 服务器上图片的地址
            "description"
            "category"
            "price"
            "start_time"    # 价格开始时间
            "end_time"      # 价格结束时间
            "unit"
        }，
        "code"
    }
}
```

## 价格相关
### 商品价格API
暂时没设置修改和删除价格，所有添加的价格都会被保留，过期的价格`status`为`False`，设为不启用。
#### 1. 添加价格
**权限：仅粮油公司组(`role=0`)**

添加价格时会将价格的`status`设置为`False`，需要经教体局账户审核后才能启用。同一个商品同时只存在有一个价格请求，后来的请求会覆盖之前的请求。
```javascript
{
    url : http://127.0.0.1:8000/api/price/
    method : POST
    data : {
        "product"           # 商品ID
        "price"             # 商品价格
        "start_time"        # 价格开始时间
        "end_time"          # 价格结束时间
    }
    return : {
        "msg",
        "data":{
            "price"
            "start_time"
            "end_time"
            "status"
            "product"
        },
        "code"
    }
}
```
示例
```javascript
{
    url : http://127.0.0.1:8000/api/price/
    method : POST
    data : {
        "product":13
        "price":7.77
        "start_time":2024-05-11
        "end_time":2024-06-11
    }
    return : {
        "msg": "Created successfully",
        "data": {
            "price": "7.77",
            "start_time": "2024-05-11T00:00:00",
            "end_time": "2024-06-11T00:00:00",
            "status": false,
            "product": 13
        },
        "code": 201
    }
}
```

```javascript

    method : POST
    return : {
        "msg"
        "data": null,
        "code"
    }
}
```

```javascript
{

    method : POST
    return : {

        "code": 200
    }
}
```
